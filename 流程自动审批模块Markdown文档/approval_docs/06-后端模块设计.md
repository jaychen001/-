## 3. 审批提交伪代码

```ts
async function submitApproval(businessType, businessId, applicantId) {
  return await db.transaction(async (tx) => {
    const businessRecord = await findBusinessRecord(businessType, businessId)

    if (!businessRecord) {
      throw new Error('BUSINESS_RECORD_NOT_FOUND')
    }

    if (!['draft', 'returned', 'withdrawn', 'rejected'].includes(businessRecord.status)) {
      throw new Error('BUSINESS_STATUS_INVALID')
    }

    const template = await tx.approvalTemplate.findFirst({
      where: {
        businessType,
        isActive: true
      },
      include: {
        nodes: {
          orderBy: {
            sortOrder: 'asc'
          }
        }
      }
    })

    if (!template) {
      throw new Error('APPROVAL_TEMPLATE_NOT_FOUND')
    }

    if (!template.nodes.length) {
      throw new Error('APPROVAL_NODE_NOT_FOUND')
    }

    const resolvedNodes = []

    for (const node of template.nodes) {
      const approver = await resolveApprover(node)

      if (!approver) {
        throw new Error('APPROVER_NOT_FOUND')
      }

      resolvedNodes.push({
        templateNode: node,
        approver
      })
    }

    const instance = await tx.approvalInstance.create({
      data: {
        templateId: template.id,
        businessType,
        businessId,
        title: generateApprovalTitle(businessType, businessRecord),
        applicantId,
        status: 'pending',
        submittedAt: new Date()
      }
    })

    let firstInstanceNodeId = null

    for (let i = 0; i < resolvedNodes.length; i++) {
      const item = resolvedNodes[i]

      const instanceNode = await tx.approvalInstanceNode.create({
        data: {
          instanceId: instance.id,
          templateNodeId: item.templateNode.id,
          nodeName: item.templateNode.nodeName,
          approverId: item.approver.id,
          status: i === 0 ? 'pending' : 'waiting',
          sortOrder: item.templateNode.sortOrder
        }
      })

      if (i === 0) {
        firstInstanceNodeId = instanceNode.id
      }
    }

    await tx.approvalInstance.update({
      where: { id: instance.id },
      data: {
        currentNodeId: firstInstanceNodeId
      }
    })

    await updateBusinessStatus(tx, businessType, businessId, {
      status: 'pending',
      latestApprovalInstanceId: instance.id
    })

    await tx.approvalLog.create({
      data: {
        instanceId: instance.id,
        businessType,
        businessId,
        operatorId: applicantId,
        action: 'submit',
        fromStatus: businessRecord.status,
        toStatus: 'pending',
        comment: '提交审批'
      }
    })

    return instance
  })
}
```

## 4. 审批通过伪代码

```ts
async function approve(instanceId, userId, comment) {
  return await db.transaction(async (tx) => {
    const instance = await tx.approvalInstance.findUnique({
      where: { id: instanceId },
      include: {
        nodes: {
          orderBy: {
            sortOrder: 'asc'
          }
        }
      }
    })

    if (!instance) {
      throw new Error('APPROVAL_INSTANCE_NOT_FOUND')
    }

    if (instance.status !== 'pending') {
      throw new Error('APPROVAL_STATUS_INVALID')
    }

    const currentNode = instance.nodes.find(node => node.status === 'pending')

    if (!currentNode) {
      throw new Error('APPROVAL_NODE_NOT_FOUND')
    }

    if (currentNode.approverId !== userId) {
      throw new Error('NOT_CURRENT_APPROVER')
    }

    await tx.approvalInstanceNode.update({
      where: { id: currentNode.id },
      data: {
        status: 'approved',
        comment,
        approvedAt: new Date()
      }
    })

    const nextNode = instance.nodes.find(node => node.status === 'waiting')

    if (nextNode) {
      await tx.approvalInstanceNode.update({
        where: { id: nextNode.id },
        data: {
          status: 'pending'
        }
      })

      await tx.approvalInstance.update({
        where: { id: instance.id },
        data: {
          currentNodeId: nextNode.id
        }
      })

      await tx.approvalLog.create({
        data: {
          instanceId: instance.id,
          nodeId: currentNode.id,
          businessType: instance.businessType,
          businessId: instance.businessId,
          operatorId: userId,
          action: 'approve',
          fromStatus: 'pending',
          toStatus: 'pending',
          comment
        }
      })

      return {
        status: 'pending',
        currentNodeId: nextNode.id
      }
    }

    await tx.approvalInstance.update({
      where: { id: instance.id },
      data: {
        status: 'approved',
        currentNodeId: null,
        completedAt: new Date()
      }
    })

    await updateBusinessStatus(tx, instance.businessType, instance.businessId, {
      status: 'approved'
    })

    await createCcUsers(tx, instance)

    await tx.approvalLog.create({
      data: {
        instanceId: instance.id,
        nodeId: currentNode.id,
        businessType: instance.businessType,
        businessId: instance.businessId,
        operatorId: userId,
        action: 'complete',
        fromStatus: 'pending',
        toStatus: 'approved',
        comment
      }
    })

    return {
      status: 'approved'
    }
  })
}
```

## 5. 审批驳回伪代码

```ts
async function reject(instanceId, userId, comment) {
  return await db.transaction(async (tx) => {
    const instance = await findPendingInstanceWithCurrentNode(tx, instanceId)

    const currentNode = instance.currentNode

    if (currentNode.approverId !== userId) {
      throw new Error('NOT_CURRENT_APPROVER')
    }

    await tx.approvalInstanceNode.update({
      where: { id: currentNode.id },
      data: {
        status: 'rejected',
        comment,
        approvedAt: new Date()
      }
    })

    await tx.approvalInstance.update({
      where: { id: instance.id },
      data: {
        status: 'rejected',
        completedAt: new Date()
      }
    })

    await updateBusinessStatus(tx, instance.businessType, instance.businessId, {
      status: 'rejected'
    })

    await tx.approvalLog.create({
      data: {
        instanceId: instance.id,
        nodeId: currentNode.id,
        businessType: instance.businessType,
        businessId: instance.businessId,
        operatorId: userId,
        action: 'reject',
        fromStatus: 'pending',
        toStatus: 'rejected',
        comment
      }
    })

    return {
      status: 'rejected'
    }
  })
}
```

## 7. 权限判断

### 7.1 是否可以编辑业务单据

```ts
function canEditBusinessRecord(record, userId) {
  if (record.applicantId !== userId) {
    return false
  }

  return ['draft', 'returned'].includes(record.status)
}
```

### 7.2 是否可以审批

```ts
function canApprove(instance, currentNode, userId) {
  return instance.status === 'pending'
    && currentNode.status === 'pending'
    && currentNode.approverId === userId
}
```

### 7.3 是否可以撤回

```ts
function canWithdraw(instance, userId) {
  return instance.applicantId === userId
    && instance.status === 'pending'
}
```

## 8. 注意事项

1. 审批流转必须使用数据库事务。
2. 审批状态和业务单据状态必须保持一致。
3. 审批日志不能删除。
4. 重新提交不建议覆盖旧审批实例。
5. 审批人解析逻辑要独立封装，方便后续支持部门负责人、项目负责人等动态审批人。
